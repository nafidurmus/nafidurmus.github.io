Cisco packet tracer faydalı komutlar

copy running-config startup-config = Çalışan yapılandırmayı başlangıç yapılandırmasını kopyala

copy run start = Çalışan yapılandırmayı başlangıç yapılandırmasını kopyala

show version = Sürümü gösterir. (Cisco paketindeki sürüm bilgileri)

show startup-config = Kopyalama izni olup olmadığını veren bootsystem var mı ?

Router> = Kullanıcı kipi istemci

Router> enable = Ayrıcalıklı kullanıcıya geçmek için gerekli komut .

Router# = Ayrıcalıklı kipi istemci

Router#disable = Ayrıcalıklı kullanıcıdan kullanıcı kipi istemciye geçiş kodu.

Router>

Router>exit = çıkış komutu

Router#configure terminal = Küresel yapılandırma kipi komutu

Router#config t = Küresel yapılandırma kipi komutu

Router(config)# = Küresel yapılandırılmış kip

Router(config)#interface fastethernet 0/1 = Arayüz yapılandırma alt kipi komutu

Router(config-if)# = Arayüz yapılandırma kipi

Router(config-if)#ip address ? = Komut aramaj için yardım işlevini kullanma(ip address 10.10.10.1 255.255.255.0)

Router(config)#router rip = Yönlendiricinin ağ adresi yönlendirme tablosu oluşturma süreci yapılandırması RIP protokolü kullanılarak yapılandırma komutu.

Router(config-router)#end = Birden fazla exit komutu kullanmadan ayrıcalıklı kipe dönme komutu

Router#

Router(config)#enable secret class = Şifre belirleme alanına giriş.

Konsol panosu

Router(config)#line console 0 = Yeni hat yapılandırma kipi

Router(config-line)#password sifre = Şifreyi sifre olarak belirledik.

Router(config-line)#login = Yönlendiriciye parola kontrolü konusunda bilgi verdik.

Router#help = Kullanılabilecek komutları listeler

Router#? = Kullanılabilecek komutları listeler

Router#con? = con__ ile başlayan komutları listeler.<cr> komutunu görünce liste bitmiştir.

% = eksik komut yazdığımızda ya da yanlış komut yazdığızda karşımıza çıkar.

Router#show history = Geçmişte kullandığımız komutları listeler (max:250)

Router#show running-config = Çalışan bütün yapılamdırmaları gösterir.

Router#show interfaces = Arayüzleri gösterir.

Router#show arp = Arp i gösterir.

Router#show ip route = ip rotasını gösterir.(c=connected, s=static , r =rip, m=mobile, b=bgp …)

Router#show protocols = Protokolleri gösterir.

Temel yapılandırma

Router(config)#hostname nafiRoute = Cihazın ismini nafiRoute olarak belirledik.

nafiRouter(config)#

Router(config)#enable password routeroute = Parolayı etkinleştirdik.

Router(config)#enable secret routeroute = Şifrelenmiş parolayı etkinleştirme.(Bu iki komut sırayla uygulanırsa alttaki parola geçerli olur.)

Router(config)#banner motd = Başlık çubuğu görünür iletisi.

Router(config)#banner login = Başlık çubuğunda oturum açma.

logging synchronous = Eş zamanlı günlüj tutma kipi

no ip domain-lookup = Ip etki alanı araması yapma.(enable ile etkinleştirdiğimizde domain-lookup ı göstermek istemediğimizde kullanılan komut.)

Sanal uç birim panosu

Router(config)#line uty 0 4 = 0–4 arası eş zamanlı bağlantıyı temsil eder.

Router(config-line)#password cisco =

Router(config-line)#login = line vty 0 gibi belirli hat bağlantı numaralarını belirleyerek her bağlantı için farklı bit parola ayarlamak mümkün.

Parola Şifreleme Gerçekleştirme

Router(config)#service password encrytion = Hizmet parola şifresi bütün parolaların şifrelenmesine yardımcı olur.

copy run start = Yaptığımız yapılandırmaları başlangıç yapılandırması dosyasına kopyalamalıyız.Yoksa yaptığımız ayarlar koybolur.
